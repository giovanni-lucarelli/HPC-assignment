\begin{frame}[fragile]{Weak Scaling}
    \begin{lstlisting}
    LOCAL_X=4096
    LOCAL_Y=4096
    N_STEPS=500

    OMP_THREADS=14        
    TASKS_PER_NODE=8
    NODE_LIST="1 2 4 8 16"      

    for NODES in NODE_LIST; do
        TOTAL_TASKS=$(( NODES * TASKS_PER_NODE ))

        case "${TOTAL_TASKS}" in
        8)    PX=4;  PY=2  ;;   # 1 node  (8 ranks)
        ...
        esac

        GRID_SIZE_X=$(( LOCAL_X * PX ))
        GRID_SIZE_Y=$(( LOCAL_Y * PY ))
        ...
    done
    \end{lstlisting}
\end{frame}
